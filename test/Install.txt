# If you want to test PROST in a Linux environment, you can install PROST in an environment with conda by running the following statement：

conda create -n PROST3.7.13 python=3.7.13
conda update -n base -c defaults conda
pip install anndata==0.7.6
pip install h5py==3.7.0
pip install leidenalg==0.9.0
pip install natsort==8.2.0
pip install numba==0.56.3
pip install numpy==1.21.6
pip install opencv-python==4.6.0.66
pip install pandas==1.3.5
pip install scanpy==1.8.2
pip install scikit-image==0.19.2
pip install scikit-learn==0.24.2
pip install scipy==1.7.3
pip install torch==1.8.1        
conda install r-base


# You also need to create an additional R language environment and install the r-mclust package, and add the R language path of this environment to the system path by running the following statement：

conda install -c conda-forge r-base=4.2.1
conda install r-mclust==5.4.10


# At this point, you can install rpy2 in the new python environment(PROST3.7.13) created in the first step:

pip install rpy2==2.9.4


# At this point in the installation, you are ready to test PROST!
